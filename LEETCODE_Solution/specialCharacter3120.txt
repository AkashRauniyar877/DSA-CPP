class Solution {
public:
    int numberOfSpecialChars(string word) {
        int count = 0;
        unordered_set<char> st(word.begin(),word.end());
        for(char ch : st){
            if(islower(ch)){
                if(st.count(toupper(ch))) count++;
            }
           
        }
        return count;
           
    }
};
