class Solution {
public:
    int reverseBits(int n) {
      vector<int>ans(32,0);
      int i = 0;
      while(n>0){
        ans[i] = n%2;
        n /=2;
        i++;
      }
    int num = 0;
    for (int i = 0; i < 32; i++) {
        num += ans[i] * pow(2, 32 - i - 1);
    }
    return num;
      
    }
};
