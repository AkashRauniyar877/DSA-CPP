class Solution {
public:
    bool hasAlternatingBits(int n) {
        string a = "";
        while(n > 0){
            int parity = n % 2;
            a += (parity + '0');   
            n = n / 2;
        }
        for(int i=0;i<a.size()-1;i++){
            if(a[i]==a[i+1]) return false;
        }
        return true;
    }
};
