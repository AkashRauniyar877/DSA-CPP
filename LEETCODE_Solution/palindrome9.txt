class Solution {
public:
    bool isPalindrome(int x) {
        if (x < 0) return false;  

        int a = x;
        long long ans = 0;

        while (x > 0) {
            int d = x % 10;
            ans = ans * 10 + d;   
            x /= 10;
        }
        if (ans == a)
            return true;
        else
            return false;
    }
};
