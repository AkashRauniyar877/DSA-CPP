class Solution{
public: 
      TreeNode* first = NULL;
    void markParent(TreeNode* root, unordered_map<TreeNode*,TreeNode*>& parent){
        if(root==NULL) return;
        if(root->left) parent[root->left] = root;
        if(root->right) parent[root->right] = root;
        markParent(root->left,parent);
        markParent(root->right,parent);
    }

    void find(TreeNode* root, int start){
        if(root==NULL) return;
        if(root->val==start){
            first = root;
            return;
        }
        find(root->left,start);
        find(root->right,start);
    }

    int amountOfTime(TreeNode* root, int start){
        find(root,start);

        unordered_map<TreeNode*,TreeNode*> parent;
        markParent(root,parent);

        queue<pair<TreeNode*,int>> q;
        unordered_set<TreeNode*> isInfected;

        isInfected.insert(first);
        q.push({first,0});

        int maxLevel = 0;

        while(q.size()>0){
            auto p = q.front();
            q.pop();

            int level = p.second;
            maxLevel = max(maxLevel,level);

            TreeNode* temp = p.first;

            if(temp->left && !isInfected.count(temp->left)){
                q.push({temp->left,level+1});
                isInfected.insert(temp->left);
            }

            if(temp->right && !isInfected.count(temp->right)){
                q.push({temp->right,level+1});
                isInfected.insert(temp->right);
            }

            if(parent.count(temp) && !isInfected.count(parent[temp])){
                q.push({parent[temp],level+1});
                isInfected.insert(parent[temp]);
            }
        }

        return maxLevel;
    }
};
