class Solution {
public:
    int digitFrequencyScore(int n) {
        unordered_map<int,int>freq;
        int sum = 0;
        while(n>0){
            int digit = n % 10;
            freq[digit]++;
            n /= 10;
        }
        for(auto x : freq){
            sum += x.first*x.second;
        }
        
        return sum;
    }
};
