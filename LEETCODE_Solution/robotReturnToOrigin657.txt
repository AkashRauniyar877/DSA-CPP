class Solution {
public:
    bool judgeCircle(string s) {
        int a = 0, b =0;
        for(int i=0;i<s.size();i++){
            char c = s[i];
            if(c=='U') a++;
            else if(c=='D') a--;
            else if(c=='R') b++;
            else b--;
        }
        if(a==0 && b==0) return true;
        else return false;
        
    }
};
